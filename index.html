Is this worth > 1000 lines of externally maintained code?

Or can we live without it? 
								
									import org.slf4j.Logger;
									import org.slf4j.LoggerFactory;
									
									class LoggerExample {
										private static final Logger log = LoggerFactory.getLogger(LoggerExample.class);
										
										void logSomething() {
											log.info("Something");
										}
									}
